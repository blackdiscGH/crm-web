

GET = DISPLAY OR SHOW A NOTE

DELETE  = SCRAP IT

POST  = WRITE A NOTE ON THE BULLETIN BOARD  (CREATED A NOEW NOTE)

PUT = ALSO CALLED PATCH


GET AND POST  ---> CHROME AND FIREFOX


POST WITH HIDDEN FIELD CALLED PUT/PATCH = PATCH ACTION  (REMEMBER FIELD IS CALLED _METHOD)
POST WITH HIDDEN FIELD CALLED DELETE = DELETE ACTION	(REMEMBER FIELD IS CALLED _METHOD)
